\begin{algorithm}[H]
    \caption{Pseudo-Code for Clustering algorithm}
    \begin{algorithmic}[1]
        \Function{AHC}{$ S, n $} \Comment{S is a set points, n is the number of final clusters}
            \STATE{$ SS \gets GenerateSuperSet(S) $} \Comment{Generate a set of sets, initially each subset has one of S's points}
            \WHILE{$ size(SS) > n $}
                \STATE{$ bestDist \gets \infty $}
                \STATE{$ bestDistPointA \gets nil $}
                \STATE{$ bestDistPointB \gets nil $} \Comment{v and p are sets}
                \FORALL{$ p \in SS $} 
                    \FORALL{$ v \in SS $}
                        \IF{$ v \neq p $ $ \AND $ $ dist(v, p) < bestDist $}
                            \STATE{$ bestDist \gets dist(v, p) $}
                            \STATE{$ bestDistPointA \gets v $}
                            \STATE{$ bestDistPointB \gets p $}
                        \ENDIF
                    \ENDFOR
                \ENDFOR
                \STATE{$ join(bestDistPointA, bestDistPointB) $}
            \ENDWHILE
        \EndFunction
    \end{algorithmic}
\end{algorithm}
